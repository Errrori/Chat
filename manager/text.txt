// 在AIService.h中添加
class StreamCallback {
public:
    virtual ~StreamCallback() = default;
    virtual void onChunk(const std::string& chunk) = 0;
    virtual void onComplete() = 0;
    virtual void onError(const std::string& error) = 0;
};

struct StreamContext {
    std::shared_ptr<StreamCallback> callback;
    std::string buffer;  // 用于处理不完整的数据块
    bool is_complete = false;
};
2. 修改WriteCallback函数
关键的修改是让WriteCallback实时处理每个数据块：
cpp
// 新的流式WriteCallback
size_t AIService::StreamWriteCallback(void* contents, size_t size, size_t nmemb, void* userp)
{
    auto* context = static_cast<StreamContext*>(userp);
    
    if (context->is_complete) {
        return size * nmemb;
    }
    
    size_t total_size = size * nmemb;
    std::string chunk(static_cast<char*>(contents), total_size);
    
    // 将新数据添加到缓冲区
    context->buffer += chunk;
    
    // 处理缓冲区中的完整数据行
    processStreamBuffer(context);
    
    return total_size;
}

void AIService::processStreamBuffer(StreamContext* context)
{
    std::string& buffer = context->buffer;
    size_t pos = 0;
    
    // 按行处理SSE数据
    while ((pos = buffer.find('\n')) != std::string::npos) {
        std::string line = buffer.substr(0, pos);
        buffer.erase(0, pos + 1);
        
        // 处理SSE格式的数据行
        if (line.starts_with("data: ")) {
            std::string data = line.substr(6);  // 去掉"data: "前缀
            
            if (data == "[DONE]" || data == "[done]") {
                context->is_complete = true;
                context->callback->onComplete();
                return;
            }
            
            // 解析JSON数据块
            try {
                Json::Value chunk_data;
                Json::Reader reader;
                if (reader.parse(data, chunk_data)) {
                    // 提取AI回复的内容
                    if (chunk_data.isMember("choices") && 
                        chunk_data["choices"].isArray() && 
                        !chunk_data["choices"].empty()) {
                        
                        const auto& choice = chunk_data["choices"][0];
                        if (choice.isMember("delta") && 
                            choice["delta"].isMember("content")) {
                            
                            std::string content = choice["delta"]["content"].asString();
                            if (!content.empty()) {
                                // 实时回调处理这个内容块
                                context->callback->onChunk(content);
                            }
                        }
                    }
                }
            } catch (const std::exception& e) {
                LOG_ERROR << "Failed to parse stream chunk: " << e.what();
            }
        }
    }
}
3. 新增流式请求方法
在AIService中添加专门的流式请求方法：
cpp
bool AIService::SendStreamRequest(const Json::Value& data,
                                 const std::string& token, 
                                 const std::string& url,
                                 std::shared_ptr<StreamCallback> callback)
{
    CURL* curl;
    CURLcode res_code;

    curl_global_init(CURL_GLOBAL_ALL);
    curl = curl_easy_init();
    
    if (!curl) {
        callback->onError("Failed to initialize CURL");
        return false;
    }

    // 设置基本选项
    curl_easy_setopt(curl, CURLOPT_URL, url.c_str());
    curl_easy_setopt(curl, CURLOPT_POST, 1L);
    
    // 构建请求体
    Json::StreamWriterBuilder builder;
    builder["indentation"] = "";
    const auto& req_body = Json::writeString(builder, data);
    curl_easy_setopt(curl, CURLOPT_POSTFIELDS, req_body.c_str());
    curl_easy_setopt(curl, CURLOPT_POSTFIELDSIZE, req_body.length());

    // 设置请求头
    struct curl_slist* headers = nullptr;
    headers = curl_slist_append(headers, ("Authorization: " + token).c_str());
    headers = curl_slist_append(headers, "Content-Type: application/json");
    headers = curl_slist_append(headers, "Accept: text/event-stream");  // 关键：接受SSE格式
    curl_easy_setopt(curl, CURLOPT_HTTPHEADER, headers);

    // 创建流式上下文
    StreamContext context;
    context.callback = callback;
    
    // 设置流式回调 - 这是关键修改！
    curl_easy_setopt(curl, CURLOPT_WRITEFUNCTION, StreamWriteCallback);
    curl_easy_setopt(curl, CURLOPT_WRITEDATA, &context);
    
    // 设置流式选项
    curl_easy_setopt(curl, CURLOPT_BUFFERSIZE, 1024L);  // 小缓冲区，更快响应
    curl_easy_setopt(curl, CURLOPT_NOPROGRESS, 0L);     // 启用进度回调
    
    // 执行请求
    res_code = curl_easy_perform(curl);

    // 清理资源
    curl_easy_cleanup(curl);
    curl_slist_free_all(headers);
    curl_global_cleanup();

    if (res_code != CURLE_OK) {
        callback->onError("CURL error: " + std::string(curl_easy_strerror(res_code)));
        return false;
    }

    return true;
}

class ChatStreamCallback : public StreamCallback {
private:
    drogon::WebSocketConnectionPtr conn_;
    int thread_id_;
    std::string accumulated_content_;
    
public:
    ChatStreamCallback(drogon::WebSocketConnectionPtr conn, int thread_id) 
        : conn_(conn), thread_id_(thread_id) {}
    
    void onChunk(const std::string& chunk) override {
        accumulated_content_ += chunk;
        
        // 实时发送给客户端
        Json::Value stream_msg;
        stream_msg["type"] = "ai_stream";
        stream_msg["thread_id"] = thread_id_;
        stream_msg["content"] = chunk;
        stream_msg["is_complete"] = false;
        
        conn_->sendJson(stream_msg);
    }
    
    void onComplete() override {
        // 发送完成信号
        Json::Value complete_msg;
        complete_msg["type"] = "ai_stream";
        complete_msg["thread_id"] = thread_id_;
        complete_msg["content"] = "";
        complete_msg["is_complete"] = true;
        complete_msg["full_content"] = accumulated_content_;
        
        conn_->sendJson(complete_msg);
        
        // 保存完整消息到数据库
        // ... 数据库保存逻辑
    }
    
    void onError(const std::string& error) override {
        Json::Value error_msg;
        error_msg["type"] = "ai_error";
        error_msg["thread_id"] = thread_id_;
        error_msg["error"] = error;
        
        conn_->sendJson(error_msg);
    }
};
